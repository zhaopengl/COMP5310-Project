import matplotlib.pyplot as plt
import numpy as np
DEFAULT_VALUE = np.nan

def clean(data, column_key, convert_function, default_value):
    special_values= {}
    for row in data:
        old_value = row[column_key]
        new_value = default_value
        try:
            if old_value in special_values.keys():
                new_value = special_values[old_value]
            else:
                new_value = convert_function(old_value)
        except (ValueError, TypeError):
            print('Replacing {} with {} in column {}'.format(row[column_key], new_value, column_key))
        row[column_key] = new_value

#Multiple Linear Regression
result=[]
index=[]
num_parameters=[]
stder_t=[]
stder_g=[]

for c in reversed(np.arange(0,0.1,0.01)):
    attributes=[]
    for i in Var_Corr:
        if abs(Var_Corr['AveragePrice']).sort_values(ascending=False)[i]>=c and i!='AveragePrice':
            attributes.append(i)
    num_parameters.append(len(attributes))
    X_data=dataframe[attributes]
    Y_data=dataframe['AveragePrice']
    X_train, X_test, Y_train, Y_test = train_test_split(X_data, Y_data, test_size=0.25,random_state=0)
    lm = LinearRegression()
    lm.fit(X_train, Y_train)
    index.append(c)
    result.append(lm.score(X_train, Y_train))
    se = math.sqrt(mean_squared_error(Y_train, lm.predict(X_train)))
    stder_t.append(se)
    se = math.sqrt(mean_squared_error(Y_test, lm.predict(X_test)))
    stder_g.append(se)

plt.plot(num_parameters, result)
plt.ylabel('Regression R-squared')
plt.xlabel('Number of parameters')
plt.show()
